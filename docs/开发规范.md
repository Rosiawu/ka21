# KA21工具导航项目开发规范

## 📖 文档概述

本文档制定了KA21工具导航项目的详细开发规范，旨在确保代码质量、维护性和团队协作效率。遵循这些规范可以有效降低技术债务，提高开发效率。

## 🏗️ 项目架构规范

### 目录结构标准

```
src/
├── app/                    # Next.js App Router页面
│   ├── globals.css        # 全局样式（仅一个文件）
│   ├── layout.tsx         # 根布局组件
│   ├── page.tsx           # 首页组件
│   ├── not-found.tsx      # 404页面
│   └── [feature]/         # 功能模块目录
│       ├── page.tsx       # 页面组件
│       └── loading.tsx    # 加载组件（可选）
├── components/            # 可复用React组件
│   ├── ui/               # 基础UI组件
│   ├── layout/           # 布局相关组件
│   └── feature/          # 功能特定组件
├── lib/                  # 工具函数和类型
│   ├── types.ts         # 全局类型定义
│   ├── utils.ts         # 通用工具函数
│   ├── config.ts        # 配置文件
│   └── validate.ts      # 数据验证函数
├── data/                 # 静态数据文件
│   ├── *.json           # JSON数据文件
│   └── *.ts             # TypeScript数据文件
├── hooks/                # 自定义React Hooks
├── utils/                # 业务相关工具函数
└── types/                # 类型定义文件
```

### 文件命名规范

- **组件文件**: PascalCase（如 `ToolCard.tsx`）
- **页面文件**: kebab-case（如 `page.tsx`, `not-found.tsx`）
- **工具函数**: camelCase（如 `sortTools.ts`）
- **类型文件**: PascalCase + `.d.ts`（如 `Team.d.ts`）
- **配置文件**: camelCase（如 `config.ts`）
- **数据文件**: kebab-case（如 `tool-categories.ts`）

1. **单一职责**: 每个组件只负责一个功能
2. **纯函数**: 避免组件内部修改外部状态
3. **Props验证**: 使用TypeScript严格类型检查
4. **可复用性**: 通过props配置化组件行为
5. **可访问性**: 遵循WCAG 2.1标准

```typescript
// 1. 纯函数：相同输入总是产生相同输出
export function calculateScore(tool: Tool): number {
  const levelScores = { high: 3, medium: 2, low: 1 };
  return levelScores[tool.recommendLevel || 'low'];
}

// 2. 单一职责：一个函数只做一件事
export function filterToolsByCategory(
  tools: Tool[], 
  category: ToolCategoryId
): Tool[] {
  return tools.filter(tool => tool.toolCategory === category);
}

// 3. 不可变性：不修改原始数据
export function addTagToTool(tool: Tool, tag: string): Tool {
  return {
    ...tool,
    tags: [...tool.tags, tag]
  };
}

// 4. 错误处理：明确的错误边界
export function parseDateSafely(dateString: string): Date | null {
  try {
    const date = new Date(dateString);
    return isNaN(date.getTime()) ? null : date;
  } catch (error) {
    console.warn(`日期解析失败: ${dateString}`, error);
    return null;
  }
}
```

```typescript
// 1. 命名：use + 功能描述
export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);
  
  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);
    
    return () => clearTimeout(timer);
  }, [value, delay]);
  
  return debouncedValue;
}

// 2. 返回对象而非数组（当返回多个值时）
```
